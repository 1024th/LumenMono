  if (n <= 1) return n;
  return fib(n - 1) + fib(n - 2);
}

From Coq Require Import ssreflect.

(* ensure proofs are well-structured *)
Set Default Goal Selector "!".
#[global] Open Scope general_if_scope.

Module list_playground.
  (* Let's do a typical proof by induction: we'll define [list] as an inductive,
     [app] (list append) as a recursive function, and prove that [app] is
     associative. *)
  Inductive list (A: Type) :=
  | nil
  | cons (x: A) (l: list A).

  (* Fix up implicit arguments. *)
  Arguments nil {A}.
  Arguments cons {A} x l.
  Notation "[]" := nil.
  Infix "::" := cons.

  Fixpoint app {A} (l1 l2: list A): list A :=
    match l1 with
    | [] => l2
    | x :: l1 => x :: app l1 l2
    end.

  Infix "++" := app.

  Theorem app_assoc {A} (l1 l2 l3: list A) :
    (l1 ++ l2) ++ l3 = l1 ++ (l2 ++ l3).
  Proof.
    induction l1 as [|x l1]; simpl.
    - reflexivity.
    - by rewrite IHl1.
  Qed.
End list_playground.

Terminal Version

“Lumen Mono Term” makes terminal symbols strictly monospaced.

Mono ~/repo  main ⇡1  git status
Term ~/repo  main ⇡1  git status

Use the regular build “Lumen Mono” for editors, and “Lumen Mono Term” for terminal applications.

Design Notes

  • Iosevka's width is 500. Lumen Mono's width is 600, a common width for monospaced fonts.
  • Metrics are adjusted to suit the 600 width.
    X-height is increased, cap height is reduced, and ascender height is increased.
  • Circular curves are rounder, like in o, e, a, s, and c.
  • Italic is based on IBM Plex Mono's humanist italic preset.
  • The neutral reading-first attitude is influenced by Commit Mono.
  • The lowercase l keeps an asymmetric serif.
    Geometric shapes for digits, like in 3, 6, and 9.
